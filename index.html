    function KasirDashboard({ products, categories, storeConfig, transactions, onLogout, onTransactionSuccess }) {

      const [cart, setCart] = useState([]);

      const [showPayment, setShowPayment] = useState(false);

      const [paymentMethod, setPaymentMethod] = useState('Cash');

      const [searchQuery, setSearchQuery] = useState('');

      const [paymentAmount, setPaymentAmount] = useState('');

      const [activeCategory, setActiveCategory] = useState('Semua');

      const [lastInvoice, setLastInvoice] = useState(null);

      const [isMobileCartOpen, setIsMobileCartOpen] = useState(false);

      const [showHistory, setShowHistory] = useState(false);

      

      const [showScanner, setShowScanner] = useState(false);

      

      const playBeep = (isSuccess) => {

        try {

          const AudioContext = window.AudioContext || window.webkitAudioContext;

          if (!AudioContext) return;

          const ctx = new AudioContext();

          const osc = ctx.createOscillator();

          const gain = ctx.createGain();

          

          osc.connect(gain);

          gain.connect(ctx.destination);

          

          if (isSuccess) {

            osc.type = 'sine';

            osc.frequency.setValueAtTime(1000, ctx.currentTime);

            gain.gain.setValueAtTime(0.1, ctx.currentTime);

            osc.start();

            osc.stop(ctx.currentTime + 0.1);

          } else {

            osc.type = 'sawtooth';

            osc.frequency.setValueAtTime(150, ctx.currentTime);

            gain.gain.setValueAtTime(0.1, ctx.currentTime);

            osc.start();

            osc.stop(ctx.currentTime + 0.4);

          }

        } catch (error) { console.error("Audio error", error); }

      };


      const filteredProducts = useMemo(() => {

        return products.filter(p => {

          const matchCat = activeCategory === 'Semua' || p.category === activeCategory;

          const matchSearch = p.name.toLowerCase().includes(searchQuery.toLowerCase()) || p.barcode === searchQuery;

          return matchCat && matchSearch;

        });

      }, [products, activeCategory, searchQuery]);


      const addToCart = (product) => {

        if (product.stock <= 0) return showAlert({ icon: 'error', title: 'Habis', text: 'Stok barang kosong!' });

        

        setCart(prev => {

          const existing = prev.find(item => item.id === product.id);

          if (existing) {

            if (existing.qty >= product.stock) {

              showAlert({ icon: 'warning', title: 'Batas', text: 'Maksimal stok tercapai.' });

              return prev;

            }

            return prev.map(item => item.id === product.id ? { ...item, qty: item.qty + 1 } : item);

          }

          const netPrice = product.price - (product.diskon || 0);

          return [ ...prev, { ...product, price: netPrice, originalPrice: product.price, qty: 1, diskon: product.diskon || 0 }];

        });

      };


      const updateQty = (id, delta) => {

        setCart(prev => prev.map(item => {

          if (item.id === id) return { ...item, qty: Math.max(1, item.qty + delta) };

          return item;

        }));

      };


      const removeFromCart = (id) => setCart(prev => prev.filter(item => item.id !== id));

      

      const totalBelanja = cart.reduce((sum, item) => sum + (item.price * item.qty), 0);


      const processPayment = () => {

        const bayar = paymentMethod === 'Cash' ? (parseInt(paymentAmount.replace(/\D/g, '')) || 0) : totalBelanja;

        

        if (paymentMethod === 'Cash' && bayar < totalBelanja) {

          return showAlert({ icon: 'error', title: 'Uang Kurang!' });

        }

        

        const transactionData = { 

          id: `INV-${Date.now().toString().slice(-6)}`, 

          date: new Date().toISOString(), 

          items: cart, 

          total: totalBelanja, 

          payment: bayar, 

          change: bayar - totalBelanja,

          method: paymentMethod

        };

        

        setLastInvoice(transactionData);

        onTransactionSuccess(transactionData);

        setCart([]); setShowPayment(false); setIsMobileCartOpen(false);

      };


      const handleSearchSubmit = (e) => {

        e.preventDefault();

        if (!searchQuery) return;

        

        const found = products.find(p => p.barcode === searchQuery || p.id === searchQuery || p.name.toLowerCase() === searchQuery.toLowerCase());

        if (found) {

           playBeep(true);

           addToCart(found);

           setSearchQuery(''); 

        } else {

           playBeep(false);

           showAlert({ icon: 'error', title: 'Tidak Ditemukan', text: `Barang dengan kode ${searchQuery} tidak terdaftar.` });

        }

      };


      useEffect(() => {

        let scanner = null;

        if (showScanner) {

          scanner = new Html5QrcodeScanner("reader", { fps: 10, qrbox: { width: 250, height: 250 } }, false);

          scanner.render(

            (decodedText) => {

              scanner.clear();

              setShowScanner(false);

              const found = products.find(p => p.barcode === decodedText || p.id === decodedText);

              if (found) {

                playBeep(true);

                addToCart(found);

                showAlert({ icon: 'success', title: 'Berhasil', text: `${found.name} ditambahkan!`, timer: 1500, showConfirmButton: false });

              } else {

                playBeep(false);

                showAlert({ icon: 'error', title: 'Tidak Ditemukan', text: `Barcode ${decodedText} tidak terdaftar di sistem!` });

              }

            },

            (err) => { }

          );

        }

        return () => {

          if (scanner) {

            scanner.clear().catch(e => console.log(e));

          }

        };

      }, [showScanner, products]);

    function PieChart({ data, labels, title, colors, isCurrency = false }) {

      const chartRef = useRef(null);

      const chartInstance = useRef(null);


      useEffect(() => {

        if (chartInstance.current) {

          chartInstance.current.destroy();

        }

        

        if (!chartRef.current) return;


        const ctx = chartRef.current.getContext('2d');

        chartInstance.current = new Chart(ctx, {

          type: 'doughnut',

          data: {

            labels: labels,

            datasets: [{

              data: data,

              backgroundColor: colors || ['#4f46e5', '#10b981', '#f59e0b', '#ec4899', '#8b5cf6', '#64748b'],

              borderWidth: 0,

              hoverOffset: 4

            }]

          },

          options: {

            responsive: true,

            maintainAspectRatio: false,

            plugins: {

              legend: { position: 'right', labels: { boxWidth: 12, usePointStyle: true, font: { size: 11 } } },

              tooltip: {

                callbacks: {

                  label: function(context) {

                    let label = context.label || '';

                    if (label) { label += ': '; }

                    if (context.parsed !== null) {

                      label += isCurrency ? new Intl.NumberFormat('id-ID', { style: 'currency', currency: 'IDR', minimumFractionDigits: 0 }).format(context.parsed) : context.parsed + ' item';

                    }

                    return label;

                  }

                }

              }

            },

            cutout: '65%'

          }

        });

        

        return () => {

          if (chartInstance.current) chartInstance.current.destroy();

        };

      }, [data, labels, title, colors, isCurrency]);


      if (!data || data.length === 0) {

        return <div className="h-[200px] flex items-center justify-center text-slate-400 text-sm italic">Belum ada data</div>;

      }


      return (

        <div className="flex flex-col h-full">

          <h3 className="font-bold text-slate-700 mb-4 text-sm">{title}</h3>

          <div className="relative flex-1 min-h-[200px]">

            <canvas ref={chartRef}></canvas>

          </div>

        </div>

      );

    }

    function AdminReport({ transactions, storeConfig }) {

        const [filter, setFilter] = useState('today');

        const [startDate, setStartDate] = useState(new Date().toISOString().split('T')[0]);

        const [endDate, setEndDate] = useState(new Date().toISOString().split('T')[0]);

        

        const filteredTransactions = useMemo(() => {

        const now = new Date();

        return transactions.filter(trx => {

          const d = new Date(trx.date);

          if (filter === 'today') return d.toDateString() === now.toDateString();

          if (filter === 'week') {

            const startOfWeek = new Date();

            startOfWeek.setDate(now.getDate() - now.getDay());

            startOfWeek.setHours(0, 0, 0, 0);

            return d >= startOfWeek;

          }

          if (filter === 'month') return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear();

          if (filter === 'custom') {

            const start = new Date(startDate);

            const end = new Date(endDate);

            start.setHours(0, 0, 0, 0);

            end.setHours(23, 59, 59, 999);

            return d >= start && d <= end;

          }

          return true;

        });

      }, [transactions, filter, startDate, endDate]);


      const calcSummary = (trxs) => trxs.reduce((acc, trx) => {

        acc.omzet += trx.total;

        trx.items.forEach(i => {

          acc.modal += (i.modal || 0) * i.qty;

          acc.laba += (i.price - (i.modal || 0)) * i.qty;

        });

        return acc;

      }, { omzet: 0, modal: 0, laba: 0 });


      const summary = calcSummary(filteredTransactions);


      const paymentStats = useMemo(() => {

        const stats = {};

        filteredTransactions.forEach(trx => {

          stats[trx.method] = (stats[trx.method] || 0) + trx.total;

        });

          labels: Object.keys(stats),

          data: Object.values(stats)

        };

          });

        const sorted = Object.entries(stats).sort((a, b) => b[1] - a[1]).slice(0, 5);

        };

      }, [filteredTransactions]);

        doc.setFontSize(16); doc.text(`Laporan Penjualan - ${storeConfig.name}`, 14, 15);

        doc.setFontSize(10); doc.text(`Periode: ${filter.toUpperCase()} | Dicetak: ${new Date().toLocaleString('id-ID')}`, 14, 22);

        });

          startY: 28, head: [['ID Transaksi', 'Tanggal', 'Metode', 'Total Modal', 'Total Omzet', 'Laba Bersih']],

          footStyles: { fillColor: [241, 245, 249], textColor: [15, 23, 42], fontStyle: 'bold' }

    function AdminProducts({ products, setProducts, categories, setCategories, syncToSheets }) {

      const [isModalOpen, setIsModalOpen] = useState(false);

      const [isCategoryModalOpen, setIsCategoryModalOpen] = useState(false);

      const [formData, setFormData] = useState({});

      const [searchQuery, setSearchQuery] = useState('');

      

      const filteredProducts = useMemo(() => {

        return products.filter(p => {

          const query = searchQuery.toLowerCase();

          return p.name.toLowerCase().includes(query) || 

                 (p.barcode && String(p.barcode).toLowerCase().includes(query)) || 

                 (p.id && String(p.id).toLowerCase().includes(query));

        });

      }, [products, searchQuery]);

      

      const [showAdminScanner, setShowAdminScanner] = useState(false);


      useEffect(() => {

        let scanner = null;

        if (showAdminScanner) {

          scanner = new Html5QrcodeScanner("admin-reader", { fps: 10, qrbox: { width: 250, height: 250 } }, false);

          scanner.render(

            (decodedText) => {

              scanner.clear();

              setShowAdminScanner(false);

              // Langsung masukin hasil scan ke inputan formData

              setFormData(prev => ({ ...prev, barcode: decodedText}));

              if (window.Swal) window.Swal.fire({ icon: 'success', title: 'Berhasil', text: `Barcode tercatat: ${decodedText}`, timer: 1500, showConfirmButton: false });

            },

            (err) => { /* abaikan error per frame */ }

          );

        }

        return () => {

          if (scanner) {

            scanner.clear().catch(e => console.log(e));

          }

        };

      }, [showAdminScanner]);


      const openModal = (p = null) => {

        setFormData(p ? { ...p} : { id: `P${Date.now().toString().slice(-6)}`, name: '', barcode: '', category: categories[0]?.name || '', stock: 0, modal: 0, price: 0, diskon: 0, image: '' });

        setIsModalOpen(true);

      };


      const handleImageUpload = (e) => {

        const file = e.target.files[0];

        if (!file) return;


        const reader = new FileReader();

        reader.onload = (event) => {

          const img = new Image();

          img.onload = () => {

            const canvas = document.createElement('canvas');

            const MAX_WIDTH = 300; 

            const MAX_HEIGHT = 300; 

            let width = img.width;

            let height = img.height;


            if (width > height) {

              if (width > MAX_WIDTH) { height *= MAX_WIDTH / width; width = MAX_WIDTH; }

            } else {

              if (height > MAX_HEIGHT) { width *= MAX_HEIGHT / height; height = MAX_HEIGHT; }

            }


            canvas.width = width;

            canvas.height = height;

            const ctx = canvas.getContext('2d');

            ctx.drawImage(img, 0, 0, width, height);


            const dataUrl = canvas.toDataURL('image/jpeg', 0.6);

            setFormData({ ...formData, image: dataUrl });

          };

          img.src = event.target.result;

        };

        reader.readAsDataURL(file);

      };


      const saveProduct = (e) => {

        e.preventDefault();

        const payload = { ...formData, stock: Number(formData.stock), modal: Number(formData.modal), price: Number(formData.price), diskon: Number(formData.diskon || 0) };

        if (products.find(p => p.id === payload.id)) {

          setProducts(products.map(p => p.id === payload.id ? payload : p));

          syncToSheets('UPDATE_PRODUCT', payload);

        } else {

          setProducts([ ...products, payload]);

          syncToSheets('ADD_PRODUCT', payload);

        }

        setIsModalOpen(false); showAlert({ icon: 'success', title: 'Tersimpan!' });

      };


      const delProd = (id) => {

        const prosesHapus = () => {

          setProducts(prev => prev.filter(p => String(p.id) !== String(id))); 

          syncToSheets('DELETE_PRODUCT', { id });

          showAlert({ icon: 'success', title: 'Produk Terhapus!', timer: 1500, showConfirmButton: false });

        };


        if (window.Swal) {

          window.Swal.fire({

            title: 'Hapus Produk?',

            text: "Yakin mau hapus? Data tidak bisa dikembalikan!",

            icon: 'warning',

            showCancelButton: true,

            confirmButtonColor: '#ef4444',

            confirmButtonText: 'Ya, Hapus!',

            cancelButtonText: 'Batal'

          }).then((result) => {

            if (result.isConfirmed) {

              prosesHapus();

            }

          });

        } else {

          if (window.confirm("Hapus produk ini? Data tidak bisa dikembalikan!")) {

            prosesHapus();

          }

        }

      };


      const addCategory = () => {

        if (window.Swal) {

          window.Swal.fire({

            title: 'Kategori Baru',

            input: 'text',

            inputPlaceholder: 'Masukkan nama kategori...',

          }).then((result) => {

            if (result.isConfirmed && result.value) {

              setCategories([...categories, newCat]); 

              syncToSheets('ADD_CATEGORY', newCat);

              showAlert({ icon: 'success', title: 'Berhasil ditambahkan!' });

            }

          });

        }

      }

            confirmButtonText: 'Ya, Hapus!',

            cancelButtonText: 'Batal'

            if (result.isConfirmed) {

              setCategories(categories.filter(c => c.id !== id));

              syncToSheets('DELETE_CATEGORY', { id });

              showAlert({ icon: 'success', title: 'Kategori Dihapus!' });

            }

          });

        }

                {/* KOLOM SEARCH BARU */}

                    className="w-full pl-10 pr-4 py-2 bg-slate-50 border border-slate-200 rounded-xl text-sm outline-none focus:border-indigo-500 font-medium transition-all shadow-inner"

    function AdminSettings({ storeConfig, setStoreConfig, syncToSheets }) {

      const [form, setForm] = useState(storeConfig);

      

      const save = (e) => { 

        e.preventDefault(); 

        setStoreConfig(form); 

        syncToSheets('UPDATE_SETTINGS', form); 

        showAlert({icon:'success', title:'Tersimpan'}); 

      };


      const handleLogoUpload = (e) => {

        const file = e.target.files[0];

        if (!file) return;

        const reader = new FileReader();

        reader.onload = (event) => {

          const img = new Image();

          img.onload = () => {

            const canvas = document.createElement('canvas');

            const MAX_SIZE = 200; 

            let width = img.width; let height = img.height;

            if (width > height) { if (width > MAX_SIZE) { height *= MAX_SIZE / width; width = MAX_SIZE; } } 

            else { if (height > MAX_SIZE) { width *= MAX_SIZE / height; height = MAX_SIZE; } }

            canvas.width = width; canvas.height = height;

            const ctx = canvas.getContext('2d');

            

            ctx.fillStyle = "#ffffff";

            ctx.fillRect(0, 0, width, height);

            

            ctx.drawImage(img, 0, 0, width, height);

            

            setForm({ ...form, logo: canvas.toDataURL('image/jpeg', 0.6) });

          };

          img.src = event.target.result;

        };

        reader.readAsDataURL(file);

      };


      return (

        <div className="bg-white p-6 rounded-2xl shadow-sm border border-slate-100 max-w-xl">

          <h2 className="text-2xl font-bold text-slate-800 mb-6">Pengaturan Toko & Struk</h2>

          <form onSubmit={save} className="space-y-4">

            

            <div className="flex items-center gap-4 mb-6">

              {form.logo ? (

                <div className="relative w-20 h-20 rounded-2xl overflow-hidden border border-slate-200 shrink-0 bg-slate-50 flex items-center justify-center p-2">

                  <img src={form.logo} className="w-full h-full object-contain" alt="Logo Preview" />

                  <button type="button" onClick={() => setForm({...form, logo: ''})} className="absolute inset-0 bg-black/60 text-white flex items-center justify-center opacity-0 hover:opacity-100 transition-opacity rounded-2xl"><Icon name="delete"/></button>

                </div>

              ) : (

                <div className="w-20 h-20 rounded-2xl bg-slate-100 border border-slate-200 flex items-center justify-center shrink-0 text-slate-400">

                  <Icon name="store" className="text-3xl" />

                </div>

              )}

              <div className="flex-1">

                <label className="block text-sm font-bold text-slate-500 mb-1">Logo Toko (Opsional)</label>

                <input type="file" accept="image/*" onChange={handleLogoUpload} className="w-full text-sm text-slate-500 file:mr-4 file:py-2 file:px-4 file:rounded-full file:border-0 file:text-xs file:font-bold file:bg-indigo-50 file:text-indigo-700 hover:file:bg-indigo-100 outline-none cursor-pointer"/>

                <p className="text-[10px] text-slate-400 mt-1">Logo ini akan muncul di bagian atas struk kasir.</p>
